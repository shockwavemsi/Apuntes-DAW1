1. Creación de la tablas

```PLSQL
-- Creación de al tabla tiendas
CREATE TABLE Tiendas (
	id_tienda NUMBER(11) PRIMARY KEY ,
	ciudad VARCHAR2(25),
	zona VARCHAR2(25)
);

-- Creación de la tabla "vendedores"
CREATE TABLE Vendedores(
	id_vendedor NUMBER(11) PRIMARY KEY,
	nombre VARCHAR2(25),
	apellidos VARCHAR(60),
	fecha_nacimiento DATE,
	cod_tienda NUMBER(11),
	fecha_contrato DATE,
	jefe NUMBER(11),
	FOREIGN KEY (cod_tienda) REFERENCES Tiendas(id_tienda),
	FOREIGN KEY (jefe) REFERENCES Vendedores(id_vendedor)
);

-- Creación de la tabla "Compradores"
CREATE TABLE Compradores(
	id_comprador NUMBER(11) PRIMARY KEY,
	nombre VARCHAR2(20),
	apellido1 VARCHAR2(20),
	apellido2 VARCHAR2(20),
	telefono VARCHAR(15),
	vendedor_que_lo_atiende NUMBER(11),
	FOREIGN KEY (vendedor_que_lo_atiende) REFERENCES Vendedores(id_vendedor)
);

-- Creación de la tabla "Articulos"
CREATE TABLE Articulos(
	id_articulo VARCHAR2(5) PRIMARY KEY,
	nombre_fabricante VARCHAR2(3),
	nombre_articulo VARCHAR2(25),
	precio_unidad NUMBER(5,2),
	existencias NUMBER(11)
);

-- Creación de la tabla "Compras"
CREATE TABLE Compras(
	num_ref_compra NUMBER(11) PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
	fecha_pedido DATE,
	comprador NUMBER(11),
	articulo VARCHAR(5),
	unidades_pedidas NUMBER(11),
	FOREIGN KEY (comprador) REFERENCES Compradores(id_comprador),
	FOREIGN KEY (articulo) REFERENCES Articulos(id_articulo)
);
```

2. Creación del procedimiento

```PLSQL
CREATE OR REPLACE PROCEDURE cantidad_euros_zona (
    -- Declaramos los parámetros de entrada
    t_zona IN VARCHAR2
)AS
    -- Declaramos las variables formales
    total_euros NUMBER (11,2);
    existe_zona NUMBER(1);

BEGIN

    SELECT COUNT(*) INTO existe_zona FROM Tiendas
    WHERE zona = t_zona;

    IF existe_zona = 0 THEN
        DBMS_OUTPUT.PUT_LINE('La zona ' || t_zona || ' no existe.');
        RETURN;
    END IF;

    SELECT SUM(c.unidades_pedidas * a.precio_unidad) INTO total_euros
    FROM Tiendas t 
        JOIN Vendedores v ON v.cod_tienda = t.id_tienda
        JOIN Compradores comp ON comp.vendedor_que_lo_atiende = v.id_vendedor
        JOIN Compras c ON c.comprador = comp.id_comprador
        JOIN Articulos a ON id_articulo = c.articulo
    WHERE t.zona = t_zona;
    
    IF total_euros IS NULL THEN
        DBMS_OUTPUT.PUT_LINE('La zona ' || t_zona || ' aún no tiene ventas');
    ELSE 
        DBMS_OUTPUT.PUT_LINE('El total de articulos vendidos en la zona ' || t_zona || ' es de ' || total_euros || '€');
    END IF;
    
END;
```
